% --- Font Families ---

\begin{frame}{Font Families}
    The theme uses the \textbf{Noto} font family by Google, chosen for its
    comprehensive Unicode coverage, consistent metrics across weights, and
    excellent on-screen legibility. Noto Sans serves as the default body font,
    with Noto Serif available for contrast and Noto Sans Mono for code listings.
    All fonts are loaded via \texttt{fontspec} and require XeLaTeX.

    \input{tex/typography/families.tex}
\end{frame}
