\begin{definition}[Continuity]
    A function $f$ is continuous at a point $c$ if
    $\lim_{x \to c} f(x) = f(c)$.
\end{definition}

\begin{theorem}[\acs{ivt}]
    If $f$ is continuous on $[a,b]$ and $d$ is between $f(a)$ and $f(b)$,
    then there exists $c \in (a,b)$ such that $f(c) = d$.
\end{theorem}

\begin{proof}
    Follows from the completeness of $\mathbb{R}$ and the
    least upper bound property.
\end{proof}
